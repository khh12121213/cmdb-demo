# 银行 CI/CD-CMDB 系统 — 部署文档

> 版本：v0.1.0 | 适用环境：银行内网 RHEL/CentOS 7+/Ubuntu 20.04+

---

## 1. 环境要求

| 组件 | 最低版本 | 说明 |
|------|---------|------|
| Python | 3.12+ | 异步协程依赖，3.10 已于 2026 年 EOL |
| MySQL | 8.0+ | 主数据存储，utf8mb4 字符集 |
| Redis | 5.0+ | 分布式锁 + 变量缓存 |
| Node.js | 18+ | 前端构建，支持 ESM |
| 操作系统 | Linux x64 | 2 核 4G 起 |

---

## 2. 基础设施准备

### 2.1 创建数据库

```sql
-- 创建数据库
CREATE DATABASE bank_cicd_cmdb
  DEFAULT CHARACTER SET utf8mb4
  DEFAULT COLLATE utf8mb4_unicode_ci;

-- 创建专用账号
CREATE USER 'cicd_admin'@'%' IDENTIFIED BY '你的密码';
GRANT ALL PRIVILEGES ON bank_cicd_cmdb.* TO 'cicd_admin'@'%';
FLUSH PRIVILEGES;
```

### 2.2 建表 DDL

> 应用启动时 SQLAlchemy `create_all` 会自动建表。以下为手动建表脚本，生产环境建议手动执行后关闭自动建表。

```sql
USE bank_cicd_cmdb;

-- 1. 环境
CREATE TABLE env_info (
  id          BIGINT        NOT NULL AUTO_INCREMENT PRIMARY KEY COMMENT '主键ID',
  env_code    VARCHAR(32)   NOT NULL COMMENT '环境编码 dev/test/staging/prod',
  env_name    VARCHAR(64)   NOT NULL COMMENT '环境名称',
  env_tags    VARCHAR(255)  NULL DEFAULT '' COMMENT '权限标签',
  is_deleted  INT           NULL DEFAULT 0  COMMENT '0正常 1删除'
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='环境信息';

-- 2. 业务系统

CREATE TABLE sys_info (

  id          BIGINT       NOT NULL AUTO_INCREMENT PRIMARY KEY COMMENT '主键',

  sys_code    VARCHAR(64)  NOT NULL COMMENT '系统唯一编码',

  sys_name    VARCHAR(128) NOT NULL COMMENT '系统名称',


  dev_owner   VARCHAR(64)  NULL DEFAULT '' COMMENT '研发负责人',

  ops_owner   VARCHAR(64)  NULL DEFAULT '' COMMENT '运维负责人',

  remark      VARCHAR(255) NULL DEFAULT '' COMMENT '备注',

  is_deleted  INT          NULL DEFAULT 0   COMMENT '0正常 1删除',

  UNIQUE KEY uk_sys_code (sys_code, is_deleted)

) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='业务系统信息';

-- 3. 应用
CREATE TABLE app_info (
  id                BIGINT        NOT NULL AUTO_INCREMENT PRIMARY KEY,
  app_code          VARCHAR(64)   NOT NULL COMMENT '应用唯一编码',
  sys_id            BIGINT        NULL DEFAULT 0   COMMENT '归属系统ID',
  app_name          VARCHAR(128)  NOT NULL COMMENT '应用名称',
  app_type          VARCHAR(32)   NOT NULL COMMENT '类型: springboot/nginx/tongweb/weblogic/tsf-service',
  repo_url          VARCHAR(255)  NULL DEFAULT ''  COMMENT '代码仓库地址',
  artifact_repo     VARCHAR(255)  NULL DEFAULT ''  COMMENT '制品仓库地址',
  owner             VARCHAR(64)   NULL DEFAULT ''  COMMENT '应用负责人',
  dev_owner         VARCHAR(64)   NULL DEFAULT '' COMMENT '研发负责人',

  ops_owner         VARCHAR(64)   NULL DEFAULT '' COMMENT '运维负责人',
  server_port       INT           NULL DEFAULT 0   COMMENT '服务端口',
  management_port   INT           NULL DEFAULT 0   COMMENT '监控端口',
  proc_name         VARCHAR(128)  NULL DEFAULT ''  COMMENT '进程名称',
  base_jvm_opts     VARCHAR(1024) NULL DEFAULT ''  COMMENT '应用全局JVM模板',
  log_base_dir      VARCHAR(255)  NULL DEFAULT ''  COMMENT '日志根目录',
  default_bk_biz_id BIGINT        NULL DEFAULT 0   COMMENT '默认蓝鲸业务ID',
  is_deleted        INT           NULL DEFAULT 0
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='应用信息';

-- 4. 集群
CREATE TABLE app_cluster (
  id               BIGINT        NOT NULL AUTO_INCREMENT PRIMARY KEY,
  env_code         VARCHAR(32)   NOT NULL,
  app_code         VARCHAR(64)   NOT NULL,
  cluster_code     VARCHAR(64)   NOT NULL COMMENT '集群唯一编码',
  cluster_name     VARCHAR(128)  NOT NULL,
  deploy_mode      VARCHAR(32)   NOT NULL COMMENT 'fixed固定主机 | elastic弹性容器',
  tsf_cluster_id   VARCHAR(128)  NULL DEFAULT '' COMMENT 'TSF平台集群ID',
  namespace        VARCHAR(64)   NULL DEFAULT '' COMMENT 'k8s命名空间',
  labels           VARCHAR(512)  NULL DEFAULT '' COMMENT '灰度发布标签',
  status           INT           NULL DEFAULT 1  COMMENT '1允许发布 0禁用投产',
  is_deleted       INT           NULL DEFAULT 0
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='应用集群';

-- 5. 部署组
CREATE TABLE app_deploy_group (
  id                          BIGINT         NOT NULL AUTO_INCREMENT PRIMARY KEY,
  env_code                    VARCHAR(32)    NOT NULL,
  app_code                    VARCHAR(64)    NOT NULL,
  cluster_id                  BIGINT         NOT NULL,
  group_code                  VARCHAR(64)    NOT NULL COMMENT '分组编码',
  group_name                  VARCHAR(128)   NOT NULL,
  deploy_group_id             VARCHAR(128)   NULL DEFAULT ''  COMMENT 'TSF平台部署组ID',
  group_type                  VARCHAR(16)    NOT NULL COMMENT 'fixed固定主机 | elastic弹性容器',
  status                      INT            NULL DEFAULT 1   COMMENT '1启用 0禁用',

  -- 虚拟机部署
  artifact_file_name          VARCHAR(128)   NULL DEFAULT ''  COMMENT '服务器固定包名(biz.tar/app.war)',
  deploy_path                 VARCHAR(255)   NULL DEFAULT ''  COMMENT '应用部署根目录',
  deploy_user                 VARCHAR(64)    NULL DEFAULT ''  COMMENT '发布执行账号',
  deploy_strategy             VARCHAR(16)    NULL DEFAULT 'full' COMMENT 'full全量 | incr增量',
  unpack_flag                 VARCHAR(1)     NULL DEFAULT 'Y' COMMENT 'Y解压部署 N不解压',
  health_check_url            VARCHAR(255)   NULL DEFAULT ''  COMMENT '应用健康检查地址',
  start_script                VARCHAR(512)   NULL DEFAULT ''  COMMENT '启动脚本',
  stop_script                 VARCHAR(512)   NULL DEFAULT ''  COMMENT '停止脚本',

  -- 通用
  jvm_opts                    VARCHAR(1024)  NULL DEFAULT ''  COMMENT '分组JVM参数(虚拟机/TSF通用)',
  package_type                VARCHAR(32)    NULL DEFAULT ''  COMMENT 'jar/war/tar',

  -- TSF弹性容器
  cpu_request                 VARCHAR(32)    NULL DEFAULT '',
  cpu_limit                   VARCHAR(32)    NULL DEFAULT '',
  mem_request                 VARCHAR(32)    NULL DEFAULT '',
  mem_limit                   VARCHAR(32)    NULL DEFAULT '',
  replicas                    INT            NULL DEFAULT 0,
  tsf_traffic_weight          INT            NULL DEFAULT 0   COMMENT '灰度流量权重',
  update_type                 INT            NULL DEFAULT 0   COMMENT '0立即更新 1滚动更新',

  -- 传统中间件
  middleware_domain           VARCHAR(128)   NULL DEFAULT ''  COMMENT '中间件域',
  middleware_cluster_name     VARCHAR(128)   NULL DEFAULT ''  COMMENT '中间件集群名',
  admin_url                   VARCHAR(255)   NULL DEFAULT ''  COMMENT '管理控制台地址',

  -- 发布并发锁
  lock_status                 INT            NULL DEFAULT 0   COMMENT '0空闲 1发布中',
  lock_trace_id               VARCHAR(128)   NULL DEFAULT '',
  lock_expire_time            DATETIME       NULL,

  is_deleted                  INT            NULL DEFAULT 0,

  UNIQUE KEY uk_group_rel (env_code, app_code, cluster_id, group_code, is_deleted),
  KEY idx_cluster_id (cluster_id)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='部署分组';

-- 6. 主机实例
CREATE TABLE cluster_instance (
  id                BIGINT        NOT NULL AUTO_INCREMENT PRIMARY KEY,
  deploy_group_id   BIGINT        NOT NULL,
  instance_ip       VARCHAR(39)   NULL DEFAULT ''  COMMENT '主机IP',

  -- 蓝鲸五元组
  bk_biz_id         BIGINT        NULL DEFAULT 0   COMMENT '蓝鲸业务ID',
  bk_host_id        BIGINT        NULL DEFAULT 0   COMMENT '蓝鲸主机ID',
  bk_cloud_id       INT           NULL DEFAULT 0   COMMENT '蓝鲸云区域ID',
  bk_module_id      BIGINT        NULL DEFAULT 0   COMMENT '蓝鲸模块ID',
  bk_inner_ip       VARCHAR(39)   NULL DEFAULT ''  COMMENT '蓝鲸管控IP',

  -- 单机差异化配置（留空则继承部署组默认值）
  ssh_port          INT           NULL DEFAULT 22,
  deploy_user       VARCHAR(64)   NULL DEFAULT ''  COMMENT '单机发布账号',
  deploy_path       VARCHAR(255)  NULL DEFAULT ''  COMMENT '单机部署路径',
  health_check_url  VARCHAR(255)  NULL DEFAULT ''  COMMENT '单机健康检查',
  start_script      VARCHAR(512)  NULL DEFAULT '',
  stop_script       VARCHAR(512)  NULL DEFAULT '',
  instance_tags     VARCHAR(255)  NULL DEFAULT ''  COMMENT '分批标签',
  instance_status   INT           NULL DEFAULT 1   COMMENT '1正常 0维护禁用',
  is_deleted        INT           NULL DEFAULT 0
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='集群主机实例';

-- 7. 变量管理
CREATE TABLE cicd_variable (
  id            BIGINT        NOT NULL AUTO_INCREMENT PRIMARY KEY,

  var_key       VARCHAR(128)  NOT NULL COMMENT '变量名',

-- 8. 部署组发布基线
CREATE TABLE app_deploy_group_release (
  id                BIGINT        NOT NULL AUTO_INCREMENT PRIMARY KEY,

CREATE TABLE cluster_instance_release (

```

### 2.3 Redis

```bash
# 安装 Redis 5.0+（已有则可跳过）
yum install -y redis  # or apt install redis-server

# 确认版本
redis-cli --version
```

## 3. 后端部署

### 3.1 获取代码

```bash
# 将 cmdb-demo 目录整体拷贝到目标服务器
scp -r cmdb-demo/ user@target-server:/data/cmdb/
cd /data/cmdb/cmdb-demo
```

### 3.2 创建虚拟环境

```bash
python3.12 -m venv venv
source venv/bin/activate
pip install -r requirements.txt
```

### 3.3 修改配置文件

编辑 `.env`：

```ini
SERVER_HOST=0.0.0.0
SERVER_PORT=8000
ENV_MODE=prod

# MySQL
DB_HOST=你的MySQL地址
DB_PORT=3306
DB_USER=cicd_admin
DB_PWD=你的密码
DB_NAME=bank_cicd_cmdb

# Redis
REDIS_HOST=127.0.0.1
REDIS_PORT=6379
REDIS_PASSWORD=
REDIS_DB=0
```

### 3.4 启动服务

```bash
source venv/bin/activate

# 前台启动（调试）
uvicorn app.main:app --host 0.0.0.0 --port 8000

# 生产启动（systemd）
# 见 5.1 节 systemd 配置
```

### 3.5 验证

```bash
curl http://127.0.0.1:8000/api/health     # → {"status":"ok"}
curl http://127.0.0.1:8000/api/ready       # → {"status":"ok","db":"connected"}
```

浏览器访问 `http://服务器IP:8000/docs` 可看到 Swagger 交互文档。

## 4. 前端部署

### 4.1 安装依赖与构建

```bash
cd /data/cmdb/cmdb-demo/frontend
npm install --registry=https://registry.npmmirror.com
npx vite build
# 产物在 dist/ 目录
```

### 4.2 单端口方案（推荐）

后端已内置 SPA 静态文件托管，`vite build` 后直接访问 `http://IP:8000` 即可，无需 Nginx / 单独前端端口。

### 4.3 Nginx 方案（可选，生产推荐）

```nginx
server {
    listen 80;
    server_name cmdb.infra.bank.com;

    root /data/cmdb/cmdb-demo/frontend/dist;
    index index.html;
    try_files $uri $uri/ /index.html;

    location /api/ {
        proxy_pass http://127.0.0.1:8000;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
    }
}
```

### 4.4 开发模式（本地/测试）

```bash
cd frontend
npx vite --host 0.0.0.0 --port 3000
# 浏览器访问 http://服务器IP:3000
```

## 5. 生产加固

### 5.1 systemd 服务单元

创建 `/etc/systemd/system/cmdb-demo.service`：

```ini
[Unit]
Description=银行 CI/CD CMDB 后端服务
After=network.target mysql.service redis.service

[Service]
Type=simple
User=appdeploy
WorkingDirectory=/data/cmdb/cmdb-demo
Environment=PATH=/data/cmdb/cmdb-demo/venv/bin:/usr/local/bin:/usr/bin
ExecStart=/data/cmdb/cmdb-demo/venv/bin/uvicorn app.main:app --host 0.0.0.0 --port 8000 --workers 4
Restart=always
RestartSec=5
LimitNOFILE=65536

[Install]
WantedBy=multi-user.target
```

```bash
systemctl daemon-reload
systemctl enable cmdb-demo
systemctl start cmdb-demo
```

### 5.2 配置安全

- 禁止将 `.env` 文件提交到 Git
- 生产数据库/Redis 密码对接内网配置中心
- 当前 demo 密码仅用于测试，**正式上线前必须更换**

### 5.3 日志

本服务使用 `loguru`（控制台输出）。生产建议配合 journald：

```bash
journalctl -u cmdb-demo -f
```

### 5.4 防火墙

```bash
firewall-cmd --add-port=8000/tcp --permanent
firewall-cmd --reload
```

### 5.5 备份建议

- MySQL：日全量备份 + binlog 增量
- 配置文件：`.env` 纳入配置管理（如 Ansible）
- 审计日志表 `sys_audit_log`：永久保留，不可物理删除

## 7. 快速排障

| 现象 | 检查 |
|------|------|
| 后端起不来 | `ss -tlnp \| grep 8000` 确认端口未被占用 |
| 数据库连不上 | `mysql -u cicd_admin -h 127.0.0.1 -p` 测试连通 |
| Redis 连不上 | `redis-cli ping` 应返回 PONG |
| 前端白屏 | 浏览器 F12 → Console 查看 JS 报错 |
| 接口 404 | 确认路径带 `/api/v1/` 前缀 |
| CORS 报错 | 后端已配置 `allow_origins=["*"]` |
